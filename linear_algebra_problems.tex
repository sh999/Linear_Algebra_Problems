	\textbf{(1.1 Treil)}
	\\\\
	Let $\vec{x} = (1,2,3)^T$,
	$\vec{y} = (y_1,y_2,y_3)^T$, 
	$\vec{z} = (4,2,1)^T.$ 
	\\
	Compute $2\vec{x}, 3\vec{y}, \vec{x}+2\vec{y}-3\vec{z}.$  	
	\\\\
	\textit{Answer}\\\\
	
	$
	2\vec{x}=2 
	\begin{bmatrix}
	1\\2\\3
	\end{bmatrix}
	= \begin{bmatrix}
	2\\4\\6
	\end{bmatrix}
	$,\\\\\\
	
	$
	2\vec{y}=2 
	\begin{bmatrix}
	y_1\\y_2\\y_3
	\end{bmatrix}
	= \begin{bmatrix}
	3\cdot y_1\\3\cdot y_2\\3\cdot y_3
	\end{bmatrix}
	$,\\\\
	
	$
	\vec{x}+2\vec{y}-3\vec{z}= 
	\begin{bmatrix}
	1\\2\\3
	\end{bmatrix}+
	\begin{bmatrix}
	2\cdot y_1\\2\cdot y_2\\2\cdot y_3
	\end{bmatrix}-
	\begin{bmatrix}
	12\\6\\3
	\end{bmatrix}=
	\begin{bmatrix}
	2y_1-11\\2y_2-4\\2y_3
	\end{bmatrix}
	$.\\
	\\\\
